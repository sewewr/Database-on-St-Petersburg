SELECT
    b.building_id,
    'Т-' || b.building_id::text,
    'technical',
    12.00 + (b.building_id % 35),
    1,
    FALSE,
    'Техническое помещение здания'
FROM buildings b;

INSERT INTO premises (building_id, premise_number, premise_type, area_m2, floor, is_residential, description)
SELECT
    b.building_id,
    'КВ-' || b.building_id::text || '-1',
    'residential',
    35.00 + (b.building_id % 60),
    CASE
        WHEN b.levels IS NULL OR b.levels < 1 THEN 1
        ELSE LEAST(b.levels, 5)
    END,
    TRUE,
    'Жилое помещение'
FROM buildings b
WHERE b.building_id % 2 = 0;

-- ============================================================
-- 5. Заполнение сервисных функций зданий
-- ============================================================

INSERT INTO building_services (building_id, service_type_id, service_status, installed_at, last_check_at, comment)
SELECT
    b.building_id,
    st.service_type_id,
    CASE
        WHEN b.building_id % 17 = 0 THEN 'repair'
        WHEN b.building_id % 19 = 0 THEN 'planned'
        ELSE 'active'
    END,
    DATE '2018-01-01' + ((b.building_id % 1500)::int),
    DATE '2026-01-01' + ((b.building_id % 120)::int),
    'Сервисная функция здания'
FROM buildings b
JOIN service_types st
    ON st.service_name IN ('электропитание', 'холодная вода', 'горячая вода', 'канализация', 'домофон');

INSERT INTO building_services (building_id, service_type_id, service_status, installed_at, last_check_at, comment)
SELECT
    b.building_id,
    st.service_type_id,
    'active',
    DATE '2018-01-01' + ((b.building_id % 1500)::int),
    DATE '2026-01-01' + ((b.building_id % 120)::int),
    'Лифт присутствует в здании с этажностью от 5 этажей'
FROM buildings b
JOIN service_types st
    ON st.service_name = 'лифт'
WHERE b.levels IS NOT NULL AND b.levels >= 5;

-- 8. Генерация заявок / задач
-- ============================================================

INSERT INTO tasks (
    building_id,
    premise_id,
    service_type_id,
    assigned_staff_id,
    task_description,
    task_status,
    created_at,
    due_at,
    completed_at,
    priority
)
SELECT
    b.building_id,
    p.premise_id,
    st.service_type_id,
    sf.staff_id,
    'Заявка на обслуживание: ' || st.service_name || ', адрес: ' || COALESCE(b.address_full, b.street, 'адрес не указан'),
    CASE
        WHEN b.building_id % 9 = 0 THEN 'new'
        WHEN b.building_id % 7 = 0 THEN 'in_progress'
        WHEN b.building_id % 5 = 0 THEN 'assigned'
        ELSE 'done'
    END,
    TIMESTAMP '2024-01-01 09:00:00' + ((b.building_id % 730) * INTERVAL '1 day'),
    TIMESTAMP '2024-01-03 18:00:00' + ((b.building_id % 730) * INTERVAL '1 day'),
    CASE
        WHEN b.building_id % 9 = 0 THEN NULL
        ELSE TIMESTAMP '2024-01-02 15:00:00' + ((b.building_id % 730) * INTERVAL '1 day')
    END,
    CASE
        WHEN st.priority_level >= 5 THEN 5
        WHEN st.priority_level = 4 THEN 4
        ELSE 3
    END
FROM buildings b
JOIN LATERAL (
    SELECT premise_id
    FROM premises p
    WHERE p.building_id = b.building_id
    ORDER BY p.premise_id
    LIMIT 1
) p ON TRUE
JOIN LATERAL (
    SELECT service_type_id, service_name, priority_level, class_id
    FROM service_types
    ORDER BY ((b.building_id + service_type_id) % 8)
    LIMIT 1
) st ON TRUE
JOIN LATERAL (
    SELECT staff_id
    FROM staff sf
    WHERE sf.class_id = st.class_id
    ORDER BY ((b.building_id + sf.staff_id) % 10), sf.rating DESC
    LIMIT 1
) sf ON TRUE
WHERE b.building_id <= 500;

-- 9. Дополнительное представление для проверки заполнения
-- ============================================================

CREATE OR REPLACE VIEW v_task_details AS
SELECT
    t.task_id,
    b.address_full,
    b.street,
    b.house_number,
    st.service_name,
    sc.class_name,
    sf.full_name AS specialist_name,
    t.task_status,
    t.created_at,
    t.due_at,
    t.completed_at,
    t.priority
FROM tasks t
JOIN buildings b ON b.building_id = t.building_id
JOIN service_types st ON st.service_type_id = t.service_type_id
JOIN service_classes sc ON sc.class_id = st.class_id
LEFT JOIN staff sf ON sf.staff_id = t.assigned_staff_id;
